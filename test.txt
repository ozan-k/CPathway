package com.ozank.cpathway.simulation;

public class ReactantPair {
    private final String name;
    private final int count;

    public ReactantPair(String name, int count) {
        this.name = name;
        this.count = count;
    }

    public String getName() {
        return name;
    }

    public int getCount() {
        return count;
    }
}
